Who do you think is happier Europeans or Americans? Hard to tell isn‚Äôt it? Let‚Äôs do some calculations

#assign name to dataframe

happiness_stats<- whr_19 %>% 
  
#remove NAs
  
na.omit(happiness_stats) %>% 

#group data by gender
  
group_by(continent) %>% 

#calculate statistics 

summarise(mean_score=mean(Score),
          
        Standard_Deviation=sd(Score),
          
        Sample_Size=n(),
          
        t_criticial=qt(0.05/2,Sample_Size-1),
          
        Standard_Error=(Standard_Deviation/sqrt(Sample_Size)),
          
        margin_of_error=Standard_Error*t_criticial,
          
        CI_high=mean_score+margin_of_error,
          
        CI_low=mean_score-margin_of_error)
